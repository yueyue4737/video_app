# config
1. npx create-react-app . --template typescript
2. yarn add @ionic/react
3. npm install -g @ionic/cli
4. ionic start my-ionic-react-app tabs --type=react
5. ionic serve
6. yarn build
7. npx cap init
8. brew reinstall cocoapods
9. npx cap add ios
10. npx cap sync ios
11. npx cap open ios
12. ionic capacitor run ios --livereload
